
#include<stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    int flag=0;
    if (n>1) {

        for (int i=2;i<n;i++) {
            if (n%i==0) {
                flag++;
                break;
            }
            else {
                flag;
            }
        }
    }
    else {
        printf("not a prime number");
    }
    if (flag==0) {
        printf("prime");
    }
    else {
        printf("not prime");
    }
    return 0;
}
