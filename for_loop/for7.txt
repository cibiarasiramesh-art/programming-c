#include<stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    int count=0;
    int add=0;
    for (int i=1;i<=n;i++) {
        int r;
        scanf("%d",&r);
        if (r>50) {
            count++;
        }
        add=add+r;
    }
    printf("total rainfall: %d\n",add);
    printf("heavy rainfall days: %d ",count);

}
