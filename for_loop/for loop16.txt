#include<stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    int or=n;
    int div;

    int add=0;
    for (int i=n;n!=0;n=n/10) {
        div=n%10;
        int fact=1;
        for (int j=div;j>0;j--) {
            fact=fact*j;

        }
        add=add+fact;

    }
    if (add==or) {
        printf("yes");
    }
    else {
        printf("no");
    }
    return 0;
}
