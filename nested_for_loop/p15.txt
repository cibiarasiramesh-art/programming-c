#include<stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    for (int row=1;row<=n;row++) {
        for (int col=1;col<=row;col++) {
            printf("*");
        }
        for (int sp=1;sp<=2*(n-row);sp++) {
            printf(" ");
        }
        for (int col=row;col>=1;col--) {
            printf("*");
        }
        printf("\n");

    }
    for (int row1=n;row1>=1;row1--) {
        for (int col1=1;col1<=row1;col1++) {
            printf("*");
        }
        for (int sp1=1;sp1<=2*(n-row1);sp1++) {
            printf(" ");
        }
        for (int col1=1;col1<=row1;col1++) {
            printf("*");
        }
        printf("\n");
    }
}
