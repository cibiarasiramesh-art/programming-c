#include<stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    for (int row=1;row<=n;row++) {
        for (int col=1;col<=n-row;col++) {
            printf(" ");
        }
        for (int sp=1;sp<=2*row-1;sp++) {
            printf("*");
        }
        printf("\n");
    }
    for (int row=n;row>0;row--) {
        for (int col=1;col<=n-row;col++) {
            printf(" ");
        }
        for (int sp=1;sp<=2*row-1;sp++) {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}
