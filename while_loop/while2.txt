#include<stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    int count=0;


    int bal;
    scanf("%d",&bal);


    while (n!=0) {
        int v;
        scanf("%d",&v);
        bal=bal+v;

        if (v<0 && bal<2000){
            count++;
            }
        n--;
    }
    printf("%d\n",bal);
    printf("%d",count);
    return 0;
}
