#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int successful = 0;
    int cancelled = 0;

    while (n != 0) {
        int v;
        scanf("%d", &v);

        if (v == 1) {
            successful++;
        } else if (v == 0) {
            cancelled++;
        }

        n--;
    }

    printf("Successful: %d\n", successful);
    printf("Cancelled: %d\n", cancelled);

    if (cancelled > successful) {
        printf("Status: Risk");
    } else {
        printf("Status: Safe");
    }

    return 0;
}
