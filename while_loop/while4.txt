#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int current = 0;
    int maxstrike = 0;

    while (n != 0) {
        int v;
        scanf("%d", &v);

        if (v == 0) {
            current++;
            if (current > maxstrike) {
                maxstrike = current;
            }
        } else {
            current = 0;
        }

        n--;
    }

    printf("%d", maxstrike);
    return 0;
}
