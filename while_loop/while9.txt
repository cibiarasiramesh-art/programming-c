#include <stdio.h>

int main() {
    int fuel, n;
    scanf("%d", &fuel);
    scanf("%d", &n);

    int i = 0;
    int tripFuel;
    int completed = 0;

    while (i < n) {
        scanf("%d", &tripFuel);

        if (fuel >= tripFuel) {
            fuel = fuel - tripFuel;
            completed++;
        } else {
            break;
        }

        i++;
    }

    printf("%d\n", completed);
    printf("%d", fuel);

    return 0;
}
