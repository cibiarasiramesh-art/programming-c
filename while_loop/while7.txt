#include <stdio.h>

int main()
{
    int capacity, N;
    int occupied = 0;
    int critical = 0;
    int i = 0;
    int x;

    scanf("%d", &capacity);
    scanf("%d", &N);

    while (i < N)
    {
        scanf("%d", &x);
        occupied = occupied + x;

        if (occupied > capacity)
            occupied = capacity;

        if (occupied >= (capacity * 90) / 100)
            critical++;

        i++;
    }

    printf("Final Occupied Beds: %d\n", occupied);
    printf("Critical Hours: %d", critical);

    return 0;
}
