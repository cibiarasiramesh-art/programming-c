#include <stdio.h>
int main() {
    int acType;
    int yrs;
    float interestRate = 0;
    scanf("%d", &acType);
    scanf("%d", &yrs);
    switch (acType) {
        case 1:   
            interestRate = 4;
            break;
        case 2:   
            if (yrs <= 3)
                interestRate = 5;
            else
                interestRate = 7;
            break;
        default:
            printf("Invalid");
            return 0;
    }
    printf("Interest Rate %.0f%%", interestRate);
    return 0;
}
