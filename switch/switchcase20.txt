#include<stdio.h>
int main(){
    int role;
    int time;
    scanf("%d%d",&role,&time);
    switch(role){
        case 1:
        printf("full access");
        break;
        case 2:
        if(time>=9 && time<=18){
            printf("Limitted access");
            break;
        }
        else{
            printf("access denied");
            break;
        }
        default:
        printf("invalid");
        break;
    }
    return 0;
}
