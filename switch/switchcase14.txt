#include<stdio.h>
int main(){
    int v,a;
    scanf("%d %d",&v,&a);
    switch(v){
        case 1:
        if(a<=5){
            printf("1500");
        }
        else if(a>5){
            printf("2500");
        }
        break;
        case 2:
        if(a<=5){
            printf("4000");
        }
        else if(a>5){
            printf("6000");
        }
        break;
        default:
        printf("invalid");
    }
    return 0;
}
