#include <stdio.h>
int main() {
    int roomType;
    char season;
    scanf("%d", &roomType);
    scanf(" %c", &season);  
    switch (roomType) {
        case 1:   
            switch (season) {
                case 'A':   
                    printf("₹2500");
                    break;
                case 'B':   
                    printf("₹2000");
                    break;
            }
            break;
        case 2:  
            switch (season) {
                case 'C':   
                    printf("₹4000");
                    break;
                case 'D':   
                    printf("₹3000");
                    break;
            }
            break;
    }

    return 0;
}
